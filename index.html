  <template #activator="{ props }">
    <v-avatar color="primary" size="30">
      <v-icon
        v-bind="props"
        size="small"
        color="white"
        class="pa-2"

        mdi-information-variant

  </template>
  {{ row.item.note }}

  v-if="row.item.note"

/>

<v-alert
  class="pa-3"
  type="info"
  color="accent"
  variant="tonal"
  density="compact"
>
  <div class="d-flex">
    <ul
      class="pl-5 my-1 align-content-center"
      :class="warnings.length === 1
        ? 'no-bullets' : ''"
    >
      <div v-for="(tenderWarning, index)
        of warnings" :key="index">
        <li>{{ tenderWarning.message }}</li>
      </div>
    </ul>
    <v-spacer />
    <v-btn
      class="align-self-end"
      variant="outlined"
      style="min-width: 100px;"
      @click="gotoTab(warnings.first().value)"
    >
      Gå til {{ warnings.first().pageName }}
    </v-btn>
  </div>
</v-alert>
Efter — DimsAlert 
<DimsAlert
  v-if="warnings.length"
  type="info"
  :alertTitle="warnings.first().message"
  :alertButtonText="`Gå til ${warnings.first().pageName}`"
  @actionClick="gotoTab(warnings.first().value)"
>
  <ul v-if="warnings.length > 1">
    <li v-for="(w, i) in warnings" :key="i">
      {{ w.message }}
    </li>
  </ul>
</DimsAlert>
